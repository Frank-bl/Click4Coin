# Install 
$ pkg update && pkg upgrade <br>
$ pkg install python git unzip curl <br>
$ git clone https://github.com/s3verus/Click4Coin.git <br>
$ cd Click4Coin <br>
$ pip install --upgrade pip <br>
$ pip3 install -r requirements.txt <br>
$ python3 main.py <br>
